ADIM 1: Temizlik, Veri Şifreleme (TDE) ve Normalizasyon
-- 1. Temizlik: Varsa eski hatalı tabloları ilişkileriyle birlikte siler
DROP TABLE IF EXISTS faturalar CASCADE;
DROP TABLE IF EXISTS musteriler CASCADE;

-- 2. Şifreleme (TDE) eklentisini aktif eder
CREATE EXTENSION IF NOT EXISTS pgcrypto;

-- 3. Müşteriler tablosunu oluşturur
CREATE TABLE musteriler (
    musteri_id INT PRIMARY KEY,
    ulke VARCHAR(50),
    sifre_hash TEXT
);

-- 4. Kirli veriyi temizleyerek (DISTINCT ON) müşterileri ve Bcrypt ile şifrelenmiş parolalarını içeri aktarır
INSERT INTO musteriler (musteri_id, ulke, sifre_hash)
SELECT DISTINCT ON (CustomerID) 
    CustomerID, 
    Country, 
    crypt('gizli_sifre_123', gen_salt('bf'))
FROM ham_veri
WHERE CustomerID IS NOT NULL;

-- 5. Faturalar tablosunu oluşturur (Müşteri ID'si ile ilişkilendirilmiş)
CREATE TABLE faturalar (
    id SERIAL PRIMARY KEY,
    fatura_no VARCHAR(20),
    stok_kodu VARCHAR(20),
    miktar INT,
    birim_fiyat DECIMAL(10,2),
    musteri_id INT REFERENCES musteriler(musteri_id)
);

-- 6. İptal/İade edilmiş faturaları ('C' ile başlayanlar) filtreleyerek temiz siparişleri aktarır
INSERT INTO faturalar (fatura_no, stok_kodu, miktar, birim_fiyat, musteri_id)
SELECT InvoiceNo, StockCode, Quantity, UnitPrice, CustomerID
FROM ham_veri
WHERE CustomerID IS NOT NULL
AND InvoiceNo NOT LIKE 'C%';


ADIM 2: Erişim Yönetimi ve Veri İzolasyonu (RLS)
-- 1. Temel Rol İzinleri: Kullanıcı sadece okuyabilir, admin tam yetkilidir
GRANT SELECT ON faturalar TO app_user;
GRANT ALL PRIVILEGES ON faturalar TO app_admin;

-- 2. Güvenlik Duvarını (RLS) Aktif Etme
ALTER TABLE faturalar ENABLE ROW LEVEL SECURITY;

-- 3. KURAL 1 (Yönetici): app_admin rolü tablodaki tüm satırları görebilir
CREATE POLICY admin_tam_erisim ON faturalar
FOR ALL TO app_admin
USING (true);

-- 4. KURAL 2 (Müşteri): app_user rolü SADECE kendi ID'si ile eşleşen faturaları görebilir
CREATE POLICY musteri_izolasyonu ON faturalar
FOR SELECT TO app_user
USING (musteri_id::text = current_setting('app.current_customer_id', true));


ADIM 3: Audit Logları (Denetim İzleri ve Tetikleyiciler)
-- 1. Güvenlik İhlallerini Kaydedeceğimiz Audit (Log) Tablosu
CREATE TABLE guvenlik_loglari (
    log_id SERIAL PRIMARY KEY,
    kullanici VARCHAR(50),
    islem_tarihi TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    islem_tipi VARCHAR(50),
    mesaj TEXT
);

-- 2. Yetkisiz Silme Denemelerini Yakalayacak Fonksiyon
CREATE OR REPLACE FUNCTION fatura_silmeyi_engelle_ve_logla()
RETURNS TRIGGER AS $$
BEGIN
    -- Eğer işlemi yapan kişi admin değilse, işlemi reddet ve logla
    IF current_user != 'app_admin' THEN
        INSERT INTO guvenlik_loglari (kullanici, islem_tipi, mesaj)
        VALUES (current_user, 'YETKİSİZ_SİLME_GİRİŞİMİ', 'Kullanıcı fatura silmeye çalıştı. Hedef Fatura No: ' || OLD.fatura_no);
        
        RAISE EXCEPTION 'Güvenlik İhlali: Müşteriler (app_user) fatura silebilir yetkisine sahip değildir!';
    END IF;
    RETURN OLD;
END;
$$ LANGUAGE plpgsql;

-- 3. Fonksiyonu Veritabanı Tablosuna Bağlayan Tetikleyici (Trigger)
CREATE TRIGGER audit_fatura_silme
BEFORE DELETE ON faturalar
FOR EACH ROW EXECUTE FUNCTION fatura_silmeyi_engelle_ve_logla();
